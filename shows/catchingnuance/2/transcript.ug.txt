ئاپتور بېككا فېرگۇسون بايان قىلغان كانادا فولكلور ۋە ئەرۋاھ ھېكايىلىرى
روبى فېرگۇسون تەرىپىدىن 1978-يىلى يۈز بەرگەن بۇ يېسىۋىلەكنىڭ تونىل ئالۋاستى
بازار ئونتارىيو
Toronto دەسلەپكى 19-ئەسىرنىڭ كۈن
بۇ يەردە نۇرغۇن دەريا ئېقىنلىرى ۋە ئېرىقلار شاخلىنىپ تۇراتتى
تومۇر ۋە ئارتېرىيە تۈگىمەس دەرەخلەر تەرەققىي قىلىۋاتقان شەھەرنىڭ ئۈستىدە ئەمەس
بۇ تېخى يېزىلاردىكى ياۋايى ھايۋانلارنىڭ دەريا ئېقىنىدا ئېگىز بىنالار ۋە بەيگىگە چۈشتى
ئالگونكۋىن ئەپسانىلىرىدە بىلىنگەن سۇ روھى بۇلارنى كۆرەلەيدۇ
ئاسان تەسكە توختايدىغان ئىنسانپەرۋەرلىك جانلىقلار ئاۋازى بار كىچىك ۋە تۈكلۈك ئىدى
شەھەر قۇرغۇچىلار يۆنىلىشكە قاراپ ماڭغاندا ئەجدىھانىڭ ئېگىز ئاۋازلىق ئۇچقۇچىسىز ئايروپىلانىغا ئوخشاش
بۇ سۇ يوللىرى يۇندى بىلەن بىرلەشتۈرۈلگەن قاتتىق يەر ئاستى تونېللىرىغا ئايلىنىدۇ
سىستېما ئاساسلىق تەڭلىمىنى بىلمەيلا دەپنە قىلىنغان دەپ قارىلىدۇ
ئۇلار 20-ئەسىرنىڭ باشلىرىدا تورونتونلىقلار ئۇزۇندىن بۇيان ئۇنتۇپ قالغان
بۇ كەڭ تونېللارنىڭ مەۋجۇتلۇقى 1978-يىلى يازنىڭ بىر كۈنىگىچە
يېسىۋىلەك بازىرى دەپ ئاتالغان شەھەرنىڭ رايونى ئېرنېست ئىسىملىك ​​بىر كىشى مەخپىي يىقىلدى
تونېلغا كىرىش ئېغىزى ئاڭلىغان يوقاپ كەتكەن مۈشۈكنى ئىزدەۋاتقاندا
ئۇنىڭ پارلامېنت كوچىسىدىكى تۇرالغۇسىنىڭ يېنىدىكى كوچىدا مىدىرلاپ ئازابلاندى
ئۇ ئارمىيەگە ئۆمىلەپ كىرىشنى قارار قىلىشتىن ئىلگىرى
تەخمىنەن ئون فۇت چوڭ چىراغ ۋە تونېل ئاستا-ئاستا كېڭەيدى
ئالۋاستىلار ئالدى تەرەپكە چېچىلىپ كەتكەن ھاڭغا ئوخشايدۇ ، ئۇ قول چىرىغىنى تىكلىدى
ئۇنىڭ مۈشۈكئېيىقنى كۆرۈشنى ئۈمىد قىلىدىغان لىم ، ئەمما ئۇ بىر جۈپ لىمغا تۇتۇلمىغان مۈشۈك ئەمەس
يانتۇ قىزىل كۆزلەر تۈكلۈك كۈلرەڭنىڭ يوغان يوغان چىشىنىڭ ئۈستىدە ئۇنىڭغا قارىدى
ئۇزۇنلۇقى تەخمىنەن ئۈچ ئىنگلىز چىسى كېلىدىغان ئىككى جانلىق بولۇپ ، بىر تەرەپكە قېچىپ كەتتى
تونېل قاتتىق ھاياجانلانغان ھالدا تونىلدىن قايتىپ كېلىپ ، بىر يىل ساقلىدى
تورونتو قۇياش گېزىتىگە ئۆزىنىڭ بۇرۇن كۆرگەنلىرىنى قوبۇل قىلىشتىن بۇرۇن
ئۇلار تونىلنىڭ كىرىش ئېغىزىنى ئىزدەپ بېرىپ ، ئۇنىڭ ئۆرۈلۈپ چۈشكەنلىكىنى بايقىغان
ئۆزى ۋە ھېچكىم بۇنداق كۆرۈشنى قايتا ئېتىراپ قىلمىغان
بۇ بېكنىڭ ئونتارىيو يېسىۋىلەك بازىرىنىڭ تونىل ئالۋاستى بولۇپ كەلدى
فېرگۇسوننىڭ كانادا فولكلورى ۋە ئەرۋاھ ھېكايىلىرى يۈرۈشلۈكلىرى تولۇق ۋە
Wattpad دا ياكى ئاپتورنىڭ تور بېكىتى becca دا بىۋاسىتە كۆپىيىۋاتىدۇ
فېرگۇسون باھاغا ئوخشاش تەمكىن بولۇپ ، مۇشتەرى بولۇشقا كاپالەتلىك قىلىڭ ، كېيىنكىسىنى قولدىن بېرىپ قويماڭ
