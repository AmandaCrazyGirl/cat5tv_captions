ئاپتور بېككا فېرگۇسون بايان قىلغان كانادا فولكلور ۋە ئەرۋاھ ھېكايىلىرى
روببى فېرگۇسون تەرىپىدىن بۇ مانىتوبانىڭ قىزىل دەرياسى ئوكسىد ئالۋاستى
قەلئەدە كۆرۈلىدىغان ئەرۋاھ
قاراۋۇللارنىڭ كېچىدە ھوشيارلىقنى ئاشكارىلاپ قويغان
1903-يىل 8-ئاينىڭ 29-كۈنىدىكى ئەتىگەنلىك تېلېگرامما گېزىتىنىڭ ماقالىسى
19-ئەسىردە ۋىننىپېگدىكى قىزىل دەريا يولى
مانىتوبا قىزىل دەريانى تۇتاشتۇرىدىغان ئوكسىگېنلىق يوللارنىڭ سودا يولى ئىدى
ئەنگىلىيە شىمالىي ئامېرىكىدىكى كولونى ۋە فورت گاررىي مىسسىسىپىغىچە
ئۇ ۋاقىتلاردا ئامېرىكىدىكى دەريا بۇ رايوندا شوتلاندىيە ئولتۇراقلاشقان
كۆچمەنلەر ۋە جۈپتى ئەينى ۋاقىتتا كەمسىتىلگەن
فرانسىيە ۋە بىرىنچى مىللەتلەرنىڭ مىراسلىرى سەۋەبىدىن يېرىم نەسىل دەپ ئاتىلىدۇ
يەر ئىككى گۇرۇپپا ئوتتۇرىسىدا ئورتاق بەھرىلەنمىگەن ، شۇڭا
فورت گاررىيدا ئوۋ ئوۋلاش گېزىتى بىرىنچى خوجايىنلارنىڭ پەرەز قىلغانلىقىنى پەرەز قىلدى
قىزىل دەريا ۋادىسى غەربىي شىمال تېغىنىڭ تاجاۋۇزىغا نارازى ئىدى
ئۆلۈكلەر ئۈچۈن مۇقەددەس مەيداندىكى مىلتىقلار ۋە ئۇلارنىڭ نارازىلىقىنى قوزغىدى
يازنىڭ كەچقۇرۇنلىرىدا بىر شۇم ياز كەچنى قاتتىق ھېس قىلدى
تۆۋەنكى تۆت ماشىنىنىڭ سىرتىدا قاراۋۇل تۇرغان ئەسكەر قاراڭغۇ بىر نەرسىنى كۆردى
كېيىنكى تۇمانغا يېقىنلاپ كېلىۋاتقان رېتىملىق قىسقۇچ كېسىش كەلدى
تۇياق تۇياق ماشىنىسى جۈپتى ھەيدىگەن قاراڭغۇلۇقتىن پەيدا بولدى
ئايال گەرچە ئەسكەرنىڭ قىسىۋالغان يەرنىڭ ۋاقتى ئۆتكەن كۆرۈنۈشىدىن گاڭگىراپ قالغان بولسىمۇ
توختاپ بىر نەچچە مىنۇتتىن كېيىن ھارۋا ئاستا-ئاستا غۇلاپ چۈشتى
يىراقتىكى تۇماندا يەنە بىر قاراڭغۇ شەكىل پەيدا بولدى
ئوخشاش بىر كالا ھارۋىسى ئۈچىنچى قېتىم ئەسكەر قوزغالغاندا قايتا سەپەر قىلدى
ئالقاننىڭ نەم ۋە تىترەپ قېشىدا ئۇ ھارۋىنى توختىتىشقا بۇيرۇدى ، ئەمما ياق
پۈتۈن كۆرۈنۈش يوقالغاندا ئۇنىڭ نوپۇزلۇق سۆزلىرى بالدۇرراق چىقىپ كەتتى
شامالدىكى ئىس-تۈتەكتەك سېكۇنتتىن كېيىن ئۇ قايتا پەيدا بولدى
ئۇ ھازىر توختاپ قېلىش ئۈچۈن ھاياجانلانغان ھالدا ۋارقىراپ يەنە غايىب بولدى
قاتتىق ۋەھىمە ئىچىدە مىلتىقنى يەرگە تاشلاپ كەينىگە قېچىپ كەتتى
قەلئە ئۇنىڭ سەپداشلىرى ئۇنىڭ ھېكايىسىدىن كۈلدى ، ئەمما ئەتىسى كەچتە يەنە بىرى
ۋەزىپە ئۆتەۋاتقان ئەسكەرنىڭ باشتىن كەچۈرگەن كەچۈرمىشلىرى بىر-بىرلەپ يۈز بەردى
مەسخىرە قىلالايدىغان ئادەم قالمىدى [مۇزىكا]
بۇ بېكا فېرگۇسوننىڭ مانىتوبا قىسمىنىڭ قىزىل دەرياسى ئوكسىد ئالۋاستى
كانادا فولكلور ۋە ئەرۋاھ ھېكايىلىرى يۈرۈشلۈكلىرى تولۇق ۋە
Wattpad دا ياكى ئاپتورنىڭ تور بېكىتى Becca دا بىۋاسىتە كۆپىيىۋاتىدۇ
فېرگۇسون ئىنكاسنى ياقتۇرىدۇ ھەمدە مۇشتەرى بولۇشنى جەزملەشتۈرۈڭ ، شۇندىلا كېيىنكىسىنى قولدىن بېرىپ قويماڭ
