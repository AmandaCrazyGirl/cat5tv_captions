ئاپتور بېككا فېرگۇسون بايان قىلغان كانادا فولكلور ۋە ئەرۋاھ ھېكايىلىرى
روببى فېرگۇسون تەرىپىدىن بۇ غەربىي شىمالدىكى ناخەننى ۋادىسىنىڭ يوقاپ كەتكەن دۇنياسى
Territories
ناخاننىي دەرياسى دەپ ئاتالغان تۇمان كېپەنلىك جىلغىدىن ئۆتىدۇ
كانادانىڭ غەربىي شىمال رايونىدىكى باشسىز ئەرلەر ۋادىسى
دۆلەت قورۇسى چوققىلارغا تەگمىگەن ئېگىز ئۆستەڭ ئۆڭكۈرلىرى بىلەن تولدى
ئورمانلىق ۋە شارقىراتما نىئاگارا شارقىراتمىسىنىڭ ئىككى ھەسسىسىگە تەڭ كېلىدۇ
سىرلىق جىلغىغا پەقەت پىيادە ياكى ئايروپىلان بىلەنلا كىرگىلى بولىدۇ
يەرلىك خەلقلەر ئاغزاكى تارىخى سۆزلەيدىغان ئوۋ ئوۋلايدۇ دەپ قارىلىدۇ
يوشۇرۇنغان مەخلۇقاتلار ھەقىقەتەن باشسىز جىلغىنىڭ ئىسمى مۇۋاپىق كېلىپ چىققان
20-ئەسىرنىڭ بېشىدىكى ئالتۇن ئالدىراش مەزگىلدە بىر قانچە سەۋەبى ئېنىق بولمىغان ئۆلۈمگە
1906-يىلى MacLeod ئاكا-ئۇكىلار ئالتۇن ئىزدەشكە ئاتلانغان ، ئەمما ئۇلارنىڭ جەسىتى
كېيىن بىر ئېرىق تەرىپىدىن تېپىلغان ئىككى كاللا كېسىلگەن باش 1917-يىلى ئەزەلدىن باشسىز تېپىلمىغان
1945-يىلى دەريادىن يېقىن جايدا سۇ ئۈزۈش ئىزدىگۈچىنىڭ جەسىتى تېپىلغان
ئونتارىيودىن كەلگەن كان ئىشچىسى ئۇخلاۋاتقان سومكىسىدىن يەنە بېشىنى تاپالمىغان
يەنە بەزىلەر بۇ ئۆلۈمنى بىر يۈرۈش قاتىلنىڭ ئەسىرى دەپ پەرەز قىلغان
باشقىلار بەك كۆپ قوشۇلمىدى ۋە ھەر قېتىملىق ئۆلتۈرۈش ئارىسىدا ئۆتۈپ كەتتى
بىر ساراڭنىڭ داناي ئىت قوۋۇرغىسى ۋە ئىنۇيىت قەبىلىسىنىڭ خىزمىتى باشقىچە ئىدى
ئەسىرلەردىن بۇيان ئۇلار زوراۋانلىق مۇسابىقىسىدىن قورقىدۇ
مايمۇنغا ئوخشاش ئىنسانپەرۋەرلىكلەر ئۇلار نوك قىياپىتى شۇنداقلا چوڭ كاناي مەخلۇق دەپ ئاتىغان
1964-يىلى ماھېلا دەپ ئاتالغان جون باپتىست قاپقىچىسى بىلەن بىللە ياۋروپالىق
تورداشلار بۇ يېڭى چاپان بىلەن قورقۇنچلۇق ئۇچرىشىشنى دوكلات قىلدى
ئۇلارنى تۈكلۈك ساقاللىق Neanderthal بەيگىسى دەپ تەسۋىرلىدى
ئېگىز موز تېرىسى كىيگەن ۋە كالتەك كۆتۈرگەن ، ئەمما گۇاجىلو تېخىمۇ كۆپ
قورقۇنچلۇق يەنىلا زېمىننى قوغدايدىغان يامان روھلار دەپ قارالدى
ئىنسانلارنىڭ تاجاۋۇزچىلىرى پەقەت زىيانكەشلىككە ئۇچرىغۇچىلارنىڭ بېشىنىلا ئېلىپ ساياھەت قىلىدۇ
قارىماققا كەڭ ئاق ۋە ئېيىقنىڭ تىرنىقى بار قار ئاق بۆرىگە ئوخشايدۇ
ئۇلار ئۇزۇندىن بۇيان يوقىلىپ كەتكەن ئېيىق ئىتقا ئوخشايدۇ
ناخەننىي جىلغىسىنىڭ يوقاپ كەتكەن دۇنيا بولۇشى مۇمكىنلىكىنى ئېيتتى
ئىگىلىك تىكلەشكە جۈرئەت قىلغانلار خېلى بۇرۇنلا ساقلانغان يادىكارلىقلار تەرىپىدىن قوغدىلىدۇ
بۇ غايەت زور زاپاسنىڭ نامەلۇم چوڭقۇرلۇقى نۇرغۇن كىشىلەر ئۈچۈن ئۆز خەۋىپىدە قىلىدۇ
كەتتى نارى قايتىپ كەلدى [مۇزىكا]
بۇ ۋان ھانى ۋادىسىنىڭ غەربىي شىمال رايونىنىڭ بىر قىسمى يوقاپ كەتكەن دۇنيا بولۇپ قالدى
بېكا فېرگۇسوننىڭ كانادا فولكلورى ۋە ئەرۋاھ ھېكايىلىرى يۈرۈشلۈكلىرى تولۇق تۇتۇلدى
ۋە Wattpad ياكى ئاپتورنىڭ تور بېكىتى بېككادا بىۋاسىتە كۆپىيىۋاتىدۇ
فېرگۇسون ئىنكاسنى ياقتۇرىدۇ ھەمدە مۇشتەرى بولۇشنى جەزملەشتۈرۈڭ ، شۇندىلا كېيىنكىسىنى قولدىن بېرىپ قويماڭ
