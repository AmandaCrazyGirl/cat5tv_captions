ئۇنى كۆچۈرۈڭ ، ئاندىن نەقىللەردە مەن روببىنى 5-تېلېۋىزورغا قويماقچى ، ئۇ
پەقەت مېنىڭ ئېلېكترونلۇق خەت ئادرېسىمنى ئېيتسام ئاندىن مەن ئىشلەتكۈچىنىڭ چېكىت نامىنى بىلمەيمەن
ئەگەر ئۇنى ئاڭلىيالىسىڭىز ، ئەمما بۇ ئەمەلىيەتتە ستۇدىيىمىزنىڭ سىرتىدىكى ھەقىقەتەن ناچار بوران
بۈگۈن چىراغلار بىر نەچچە قېتىم لەپىلدەپ تۇرماقتا ، ئەمما مېنىڭچە ئېرىشىمىز دەپ ئويلايمەن
قانداقلا بولمىسۇن بۇ ئارقىلىق ئۇنى ئۆچۈرۈۋېتىڭ ۋە ئىسمىمنى روببىغا ئۆزگەرتىڭ
فېرگۇسون ماقۇل بولدى ، شۇڭا مەن پەقەت بىرلا قېتىم قىلىشىم كېرەك
كېلەر قېتىم مەن بۇ git ۋەدىسىنى ئىجرا قىلغىنىمدا - ئۇنى قوشۇۋاتامسىز؟
شۇڭا ئۇنى قوشۇۋالدىم ، ئۇ بىر ھۆججەتنىڭ ئۆزگەرگەنلىكىنى كۆرىدۇ
قىستۇرما ۋە SH نى سىناق قىلىش يېڭى ، شۇڭا ئۇ ئەڭ ئاخىرقىسىنى بارلىققا كەلتۈرىدۇ
مەن كىرگۈزۈشكە تېگىشلىك بۇيرۇق بولسا ئىتتىرىش مەنبەسىگە ئېرىشىش ، مەن بۇنى بەلگىلەيمەن
ئۇستاز شاخ ئۇستىسى بولغاچقا كېلىپ چىققان ئۇستاز ، ھازىر ئۇ بارىدۇ
مەن قۇرغاندا كىرگۈزگەن ئىشلەتكۈچى ئىسمىم ۋە پارولىمنى سوراڭ
ھېسابات شۇڭا cat5 TV مېنىڭ ئىشلەتكۈچى ئىسمىم ۋە مەخپىي نومۇرىم LastPass نى ئىشلىتىپ يېڭى ھاسىل قىلىمەن
پارول ھەر ۋاقىت ، ئۇلار كەڭ ۋە ساراڭ بولغاچقا مەن پارولنى كۆچۈردۈم
ئاندىن ئۇنى چاپلىدىم ، ئۇ يەرگە بېرىپ ئۇ سانلىق مەلۇماتلارنى ماڭا يۈكلەيدۇ
ئامبار ۋە ئۇ يېڭىلاش ئۈچۈن f5 قىلىپ ئىشلەندى ، سىز SH سىنى سىناپ بېقىشىڭىز كېرەك
ھازىر مېنىڭ github ئامبىرىمنىڭ بىر قىسمى بولۇپ ، ئۇ بار ۋە ئۇنى كۆرسىتىپ بېرىدۇ
ئىجرائىيە سالام دۇنيا شۇڭا ھازىر بۇ يەرگە قايتىپ كېلىڭ ، مەن پەقەت ئۆزگەرتىش كىرگۈزگەندەك قىلايلى
باشقا بىر كومپيۇتېردا باشقا كومپيۇتېردا مەن توركۆرگۈچ ئارقىلىق شۇنداق قىلىمەن
بۇ سىستېمىدا سىزگە كۆرسىتىپ قوياي ، مەن يەنە بىر قېتىم ياخشىمۇسىز
مەن بۇ ئۆزگىرىشلەرنى ساقلايمەن ، ئۇنىڭغا ئىسىم قويمايمەن ياكى ئۇنىڭغا ئوخشىمايمەن
بۇ تەسۋىرنى مەن جەزىملەشتۈرەلەيمەن ، ھازىر قوليازما ئوخشايدۇ
توركۆرگۈدە ئۇنى قانداق ئۆزگەرتىمەن ، بۇنى ھازىر خالىغان يەردىن قىلالايسىز
مەن ئۇ يەردىكى ئامباردىكى كومپيۇتېرىمدىكى ھۆججەتلىرىمنى يەرلىككە قارايمەن
پەقەت ياخشىمۇسىز دۇنيا دەيدۇ ، شۇڭا مەن يەنە تارتماقچى ، مەن بۇلارنىڭ ھەممىسىنى قىلىۋاتىمەن
