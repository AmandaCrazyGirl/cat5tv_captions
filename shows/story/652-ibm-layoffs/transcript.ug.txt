ھەپتىلىك مەزمۇنلارنى ئۆز ئىچىگە ئالغان Hewlett Packard لىنۇكىسقا ئوخشاش بىر تەرەپلىمە قاراش
كارخانا ۋە IBM كۆرۈنەرلىك تەننەرخنى تۆۋەنلىتىش تەدبىرلىرىنى ئېلان قىلدى
مائاشىنى كېمەيتىش ۋە كوۋا 19 كرىزىسى ئېغىر خىزمەت زىيىنىنى ئۆز ئىچىگە ئالىدۇ
ھەممە بازار ساھەسىدە دېگۈدەك جاپالىق ، ھازىر دومىنولار چۈشۈشكە باشلىدى
باشقا ئوتتۇرا ، چوڭ كارخانىلار تىجارەتنى ئازايتىدۇ ياكى تەۋرىنىدۇ
كارخانا خېرىدارلىرىغا تايىنىدىغان تېخنىكا شىركەتلىرىنىڭ ئۆزى ئېغىر
IBM شىركىتى پەيشەنبە كۈنى كەچتە خادىم قىسقارتىدىغانلىقىنى ئېلان قىلدى
شىركەت باياناتىدا رىقابەت كۈچى يۇقىرى بازار تەلەپ قىلىدىغانلىقىنى ئوتتۇرىغا قويدى
بۇ خىل ئەھۋالدا يۇقىرى قىممەتلىك ماھارەتلەرنى توختىماي رېمونت قىلىشقا جانلىق
دېمەك ، سىز ئەمدى مۇھىم ماھارەتكە يۇقىرى باھا قويماسلىقنى قارار قىلىڭ
خىزمەتچىلەرنىڭ سانى IBM غا ئوخشاش ئىجتىمائىي ئارىلىق جەدۋىلىگە ئېلىپ كېلىدۇ
ھازىر قىسقارتىش ۋە خادىم قىسقارتىشقا دۇچ كەلگەن شىركەتلەر ئەڭ ياخشى ئىقتىسادىي ۋەزىيەتتە ئەمەس
كوۋا 19 نى سوقۇشتىن بۇرۇن شىركەتنىڭ باش ئىجرائىيە ئەمەلدارى ئارۋىند كىرىشنا بۇ شىركەتتە بولغان
نەچچە ئون يىل مابەينىدە پەقەت 4-ئايدىكى ئەڭ يۇقىرى ئۇرۇققا قەدەم قويدى
شىركەتنىڭ بۇلۇت ھېسابلاشنى قوللايدىغان قىسىملىرىنى قۇرۇشقا ئەھمىيەت بەرگەن
ھەمدە سۈنئىي ئىدراك بولۇپ ، قالغان IVM دىن يىراقلىشىشنى خالايدۇ
قانچە ئورۇننىڭ قىسقارتىلىۋاتقانلىقىنى ئېنىق ئېيتمىدى ، ئەمما ھەر ئىككىسى ۋال كوچىسى
ژۇرنال ۋە بىلۇمبېرگ خەۋەرلىرى مىڭلىغان خىزمەتچىلەرنىڭ تەسىرگە ئۇچرىغانلىقىنى خەۋەر قىلدى
بەش شىتات كالىفورنىيە نيۇ-يورك شىمالىي كارولىنا مىسسۇرى ۋە پېنسىلۋانىيە IBM
باياناتىدا ئاپەتكە ئۇچرىغانلارغا تولۇقلىما داۋالاش كاپالىتى بىلەن تەمىنلەيدىغانلىقىنى ئېيتتى
كەلگۈسى 12 ئايدىكى خىزمەتچىلەرمۇ Hewlett Packard Enterprise شىركىتىمۇ بۇنى ئېلان قىلدى
پەيشەنبە كۈنى يېقىنقى پەسىللىك كىرىمىنىڭ بىر قىسمى سۈپىتىدە تەننەرخنى تۆۋەنلىتىش پىلانى
دوكلاتتا دېيىلىشىچە ، شىركەت ھېچ بولمىغاندا ئاخىرىغىچە بىر قىسىم ئىش ھەققىنى قىسقارتىدىكەن
ئۆكتەبىردە يۇقىرى دەرىجىلىك باشقۇرغۇچىلار شىركەتنىڭ مائاشىنىڭ% 20 تىن% 25 كىچە تۆۋەنلىگەنلىكىنى كۆردى
