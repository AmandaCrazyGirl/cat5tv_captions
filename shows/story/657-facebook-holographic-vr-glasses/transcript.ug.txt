Linux نىڭ بىر تەرەپلىمىلىك Facebook قا ئوخشاش ھەپتىلىك دەرسلىكلىرىنى ئۆز ئىچىگە ئالىدۇ
مەۋھۇم رېئال تىڭشىغۇچ ئۇششاق كۆزەينەكتىن كۆپ چوڭ ئەمەس
كەلگۈسى سايىسى ئالاھىدە لايىھەلەنگەن گولوگرافىك فىلىمنى ئىشلىتىپ كىچىكلىتىدۇ
لىنزا ئادەتتىكى VR كۆرسەتكۈچلىرى سۇندۇرۇش ئەينىكىدەك چوڭ بولىدۇ
ئۇلارنىڭ ئىچىگە ئېكراننى مەركەزلەشتۈرۈش ئۈچۈن بىر نەچچە ماتور دىيۇم لازىم
تاقىغۇچىلارنىڭ كۆزى بىر جۈپ كۆزئەينەك بولۇپ ، ئەڭ كۆپ بولغاندا ئاران 9
تەتقىقاتچىلار ۋە ئۇسۇلغا ئاساسەن قېلىنلىقى مىللىمېتىر
كۆزئەينەكتىن پەقەت 17.8 گرام رەسىم يېشىل ۋە قارا ئېكران ئوچۇق
پەۋقۇلئاددە تور كۆزەينىكى تەخمىنەن 90 گرادۇس بىلەن تەمىنلەيدۇ
ئېلان قىلىنغان ماقالىدە تەپسىلىي سىناققا ئاساسەن گورىزونتال كۆرۈنۈش مەيدانى
Facebook بۇ لايىھەنى ئەنئەنىۋى سۇيۇق كرىستاللىق تاختا بىلەنمۇ يوقىتىدۇ
ئادەتتىكى VR تىڭشىغۇچتا ئىشلىتىلىدۇ ھەمدە ئۇنىڭ ئورنىغا لازېر ئىشلىتىپ رەسىم ھاسىل قىلىدۇ
دېمەك ، پېكسىل ساناش ئەمەلىيەتتە بىر تاللاش ئەمەس
نۇرغۇن چەكلىمىلەر بىلەن بۇ يورۇق نۇر ئارقا شامالدۇرغۇچتىن ئۆچۈرۈلىدۇ
ئۇلار گولوگرافىك نۇر دەستىسى بۆلۈش يۈزىگە مەركەزلەشكەن
كۆرسىتىشنىڭ چوڭ سەۋەبلىرى كۆرسىتىش رايونىغا تۆھپە قوشمايدۇ
تەتقىقاتچىلارنىڭ قەغەزگە يازغان كۆزئەينىكىنى كۆرۈش يەنىلا كۆزەينەك ئەمەس
تولۇق رەڭلىك رەسىم ھاسىل قىلىپ ، گىرۋەكلىرىنى چۆرىدىگەن ھالدا ئەرۋاھلارنى كۆرسىتىدۇ
ئوپتىك يۈزى كەلتۈرۈپ چىقارغان كۆرۈنۈش مەيدانى ئەترەتنىڭ ئاللىبۇرۇن يورۇقلۇقنى ئەكس ئەتتۈرىدۇ
سۇلياۋ ئاستىغا ئالماشتۇرۇش ئارقىلىق ئېغىرلىقنى تېخىمۇ تۆۋەنلىتىشنى پىلانلاڭ
تەتقىقاتچىلار ئومۇمىي ئېغىرلىقى ئاران 6.6 گرامغا يەتكۈزۈشنى ئۈمىد قىلدى
بىر جۈپ چوڭ ئاۋىئاتسىيە ئۇسلۇبىدىكى كۆزەينەك بۇ خەۋەر ئوخشاش ۋاقىتتا كېلىدۇ
گۇگۇلنىڭ تەرەققىي قىلىۋاتقان كانادا ئەقلىي ئىقتىدارلىق كۆزئەينەك شىركىتى سېتىۋالغان
يېنىك فوكۇسلۇق ئەقلىي ئىقتىدارلىق كۆزئەينەك گولوگرافىك ئېكران بىلەن لايىھەلەنگەن
ئىشلەتكۈچىلەر ئۇنىڭ كۈچەيتىلگەن رېئاللىق كۆرۈنگەنلىكىنى كۆرەلەيدۇ
قىزىقارلىق تېخنىكىدىن دېرەك بېرەلەيدىغان ئىككى چوڭ قويغۇچ
