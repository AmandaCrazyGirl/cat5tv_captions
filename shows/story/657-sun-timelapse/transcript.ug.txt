لىنۇكىس بىر تەرەپلىمىلىكى قاتارلىق ھەپتىلىك دەرسلىكلەرنى ئۆز ئىچىگە ئالىدۇ
قۇياش خەتەرلىك ، ئەمما بۇ ئاي NASA نىڭ قۇياش ئېنېرگىيىسىنىڭ ئون يىللىقىنى خاتىرىلەيدۇ
رەسەتخانا بىزگە قۇياشقا تىكىلىپ تۇرغان بۇ يۇلتۇزنى ئۆگىنىۋاتىدۇ
يەرشارىغا ئەڭ يېقىن ئوربىتىدىن 425 مىليون يۇقىرى ئېنىقلىقتىكى يىغىلىشتىن توختىمايدۇ
2010-يىلى 6-ئايدىن باشلاپ توپلانغان سانلىق مەلۇماتلار ئالىملارنىڭ ياسىشىغا ياردەم بەردى
نۇرغۇن بايقاشلار ، ئەمما بىز ئۈچۈن تورنى مەلۇم خىلغا ئايلاندۇرىدىغانلار ئۈچۈن
خۇشخەۋەر nASA كۆزىتىشلەرنى قىزىقارلىق nASA غا ئايلاندۇردى
رەسىملەرنى يۇلتۇزلار پائالىيىتىنىڭ كىشىنى ھەيران قالدۇرىدىغان ۋاقىت ھالقىسىغا ئايلاندۇردى
يېقىنقى 10 يىلدا ۋاقىت ھەددىدىن زىيادە تارتىلغان سۈرەتلەرنى قۇراشتۇرىدۇ
ئۇلترا بىنەپشە دولقۇن ئۇزۇنلۇقى قۇياشنىڭ ئەڭ سىرتقى ئاتموسفېرا قاتلىمىنى كۆرسىتىدۇ
كورونا كىنوسى ھەر سائەتتە قاراڭغۇ تام تەسۋىر بىلەن رەسىم ھاسىل قىلىدۇ
يەر شارى ياكى ئاي كۆزىتىش ئەسۋابىنىڭ ئارىسىدىن ئۆتكەندە تۇتۇلدى
ئالەم كېمىسى ۋە قۇياشنىڭ تولۇق فىلىمى 61 مىنۇت داۋاملىشىدۇ
قۇياشنىڭ ئايلىنىشى ۋە پائالىيىتىنىڭ تۆۋەنلىشى بىلەن 11 يىللىق قۇياش دەۋرى
ئىششىق قۇياش داغلىرىنىڭ كۆپىيىشىگە ئەگىشىپ پائالىيەتنى ئېنىق كۆرگىلى بولىدۇ
ماگنىت مەيدانى ۋە قۇياش نۇرىنىڭ شىددەتلىك قامچىلىرى بىلەن پارتىلاش دەرىجىسىگە
ئاندىن ئۇ يەنە بىر قېتىم تىنچلىنىدۇ ، مىقدارى بولغاندا قۇياشنىڭ ئەڭ تۆۋەن چېكى دەپ ئاتىلىدۇ
قۇياش پائالىيىتى بىر قەدەر تۆۋەن ، بۇ بىز ھازىر تۇرۇۋاتقان مەزگىل
نەتىجىدە شىمالىي چىراغ تۆۋەنرەك كۆرۈلىدۇ
كەڭلىك ئون يىل بولسىمۇ بىزنىڭ كۆز قارىشىمىزدىن كۆرۈش ئادەمنى ھەيران قالدۇرىدۇ
بىر چولپاننىڭ ھاياتىدىكى بىر قاپلىنىشتىن باشقا ، 4k لىق پۈتۈن ۋاقىتتىكى سىن
ئېنىقلىق دەرىجىسىنى cat5 چېكىتلىك تېلېۋىزورنىڭ قۇياش نۇرىدا كۆرگىلى بولىدۇ
