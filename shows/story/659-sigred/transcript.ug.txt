ھەپتىلىك ئەڭ يۇقىرى تېخنىكا ھېكايىلىرىنى ئازراق linux bias microsoft بىلەن قاپلاش
Windows مۇلازىمېتىر خېرىدارلىرىغا بۇ يوچۇقنى ياماق قىلىشنى جىددىي تەۋسىيە قىلىش
ھۇجۇم قىلغۇچىلارنىڭ پۈتكۈل تورلارنى كونترول قىلىشىغا يول قويىدۇ
ئىشلەتكۈچىنىڭ ئۆز-ئارا تەسىر قىلىشى يوق ھەمدە ئۇ يەردىن كومپيۇتېردىن كومپيۇتېرغا تېز تارقىلىدۇ
بۇ بايقاشنى تەكشۈرۈش پونكىتىدىكى تەتقىقاتچىلار تاماكا دەپ ئاتىغان
ئۇ windows dns نىڭ تەركىبىدە تۇرىدۇ
تور نامىنى ipi غا تەرجىمە قىلىش تەلىپىگە ئاپتوماتىك جاۋاب قايتۇرىدۇ
ئادرېس كومپيۇتېرلىرى ئۇنى توردىن تېپىشى كېرەك
يامان غەرەزدە شەكىللەنگەن سوئاللارنى ئەۋەتىش ئارقىلىق ھۇجۇم قىلغۇچىلار ئىجرا قىلالايدۇ
دائىرە باشقۇرغۇچى ھوقۇقىغا ئېرىشىدىغان كود ، ئاندىن ئۇ يەردىن
پۈتكۈل تورنى كونترول قىلىڭ ، بارلىق كۆزنەكلەردە يوچۇق بار
مۇلازىمېتىر نۇسخىسى 2003-يىلدىن 2019-يىلغىچە. مىكروسوفت ۋە
تەتقىقاتچىلار ئۇنىڭ كومپيۇتېردىن تارقىلىشى مۇمكىن بولىدىغان قۇرتلۇق مەنىنى ئوتتۇرىغا قويدى
كومپيۇتېر ئىشلەتكۈچىلەر يوق دومىنونىڭ چۈشۈپ كېتىشىگە ئوخشايدۇ
ئۆز-ئارا تەسىر كۆرسىتىدىغان كومپيۇتېر قۇرتلىرىنىڭ يوشۇرۇن كۈچى بار
ئۇلىنىش ئارقىلىقلا تېز تەشۋىق قىلىڭ
ھەمدە ئاخىرقى ئىشلەتكۈچىلەردىن ھېچ ئىش قىلىشنى تەلەپ قىلمايلا
بىر قۇرتنىڭ يوشۇرۇن يوچۇقلىرى ئاسانلا يامان غەرەزلىك كودلارنىڭ بولۇشىغا يول قويىدۇ
ئىجرا قىلىنغان ئېكىسپېدىتسىيە ئالاھىدە زىيانلىق بولىدۇ
2016-يىلدىن باشلاپ پېچا ھۇجۇمى ئەمەس ، ھەم خاھىشتا يۈز بەرگەن
دۇنيا مىقياسىدا تورنى تاقاپ ، مىليارد دوللار زىيان كەلتۈرۈپ چىقاردى
تەكشۈرۈش پونكىتى تەتقىقاتچىلىرى سىگ قىزىلنى ئىشلىتىش ئۈچۈن تەلەپ قىلىنغان تىرىشچانلىقنىڭ بارلىقىنى ئېيتتى
ماھارەتلىك خاككېرلارنىڭ ۋاستىسى ئىچىدە ، بۇنىڭ ھېچقانداق پاكىت يوق
ھازىرچە ئاجىزلىق ئاكتىپلىق بىلەن ئىشلىتىلمەيدۇ
تەكشۈرۈش پونكىتى ھەمدە بۇنىڭ ئۆزگىرىشى مۇمكىنلىكىنى ، ئەگەر بۇزغۇنچىلىق قىلىدىغان بولسا دېدى
ئۈنۈمى يۇقىرى مىكروسوفتنىڭ مۇمكىنچىلىكىنى باھالايدۇ
ئېكسپىلاتاتسىيە قىلىش ئېھتىماللىقى نۇرغۇن سىرتقى تەتقىقاتچىلار
بىخەتەرلىك رېئاكتورى تەتقىقاتچىسى ماركۇس خۇچىننىڭ ئەندىشىسى
ھۇجۇم قىلغۇچىلار تاماكا چەككەندىن كېيىن مېيىپ بولۇپ قالىدۇ
بۇ سىنارىيەدىكى ھۇجۇم قىلغۇچىلار ئېلىپ بارىدىغان تۆلەم يۇمشاق دېتالى
تورنىڭ dns مۇلازىمېتىرىنى كونترول قىلىڭ ، ئاندىن ئۇنى ئىشلىتىپ يامان غەرەزلىك يۇمشاق دېتالنى كۆپچىلىككە ئىتتىرىڭ
ئۇلانغان خېرىدارلار كومپيۇتېرى microsoft بۇنىڭ بىر قىسمى سۈپىتىدە ئوڭشاشنى ئېلان قىلدى
windows dns ئىشلىتىدىغان ئايلىق يېڭىلاش سەيشەنبە تەشكىلاتلىرى
خەتەرنى ئەستايىدىللىق بىلەن باھالاپ ، سەيشەنبە كۈنىدىكى ياماقنى تېزرەك ئورنىتىشى كېرەك
